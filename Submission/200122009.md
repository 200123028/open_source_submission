## Code

#include <iostream>
using namespace std;

int main() {
    string name;
    int sum=0;

    cout << "Enter your Name: ";
    getline(cin, name);

    int len = name.length();

    for(int i=0; i<len; i++){
        sum += int(name[i]);
        }

    cout << "Sum of ASCII Value of all Characters(Including Spaces) of " << name << " is: " << sum;

    return 0;
}
