### Program to print sum of ASCII characters
```cpp
#include <iostream>
using namespace std;
int main(){
	char name[]="Sarthak Diwan"; //Pascal Case
	int sum=0;
	for(int i=0;name[i]!='\0';i++){
		sum+=(int)name[i];
	}
	cout<<sum<<endl;
	// To remove space
	// cout<<sum-32<<endl;
	return 0;
}
```
