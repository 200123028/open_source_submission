### C++ code to find the ASCII sum of a string (My Name)

```
C++
#include<iostream>
#include<string>

using namespace std;

int main ()
{
    string name;
    int x = 0; 
    name = "Prem Kumar";
    for (int i = 0; i < name.size(); i++)
    {
        int t = name[i];
        x = x + t;
    }
    cout << x;
    return 0;
}
```
