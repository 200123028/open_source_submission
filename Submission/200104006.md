**CODE FOR SUM OF ASCII VALUES OF 'MY NAME'**

```cpp
#include <iostream>
using namespace std;

int main() {
	
	string s = "ADITYA RAJ";
	int sum = 0;
	for(int i = 0; i < s.length(); i++)
	{
	    if(s.at(i)==' ')
	    continue;
	    else
	    sum = sum + (int)s.at(i);
	}
	cout << "Sum of ASCII values of My Name: " << sum << endl;
 	return 0;
}
```
