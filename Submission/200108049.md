### C++ code for sum of ASCII value of string :
```C++
#include <iostream>
using namespace std;

int main(){

    string fullName = "Shikhar Bind";
    int asciiSum = 0;
    int length = fullName.length();

    for (int i = 0; i < length; i++)
    {
        asciiSum += fullName[i];
    }

    cout<<"The sum of ASCII values of the Full Name is "<<asciiSum<<"."<<endl;
    
    
    return 0;
}
```
