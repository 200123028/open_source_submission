### C++ code to find the ASCII sum of a string

```
C++
#include<iostream>
#include<string>
using namespace std;
int main ()
{
    string name;
    int sm = 0; 
    name = "Shreyans Khabia";
    for (int i = 0; i < name.size(); i++)
    {
        sm = sm + name[i];
    }
    cout << sm;
    return 0;
}
```
