## C++ code to find sum of ASCII value of MY NAME

``` cpp
#include<iostream>
#include<cstring>
using namespace std;
int main()
{ 
  int sum; 
  char name[40]="NAVEEN VISHWAKARMA";
  int j=strlen(name);
  for(int i=0;i<j;i++)
   {
    sum+=int(name[i]);  
   }
  cout<<"sum of ASCII value of MY NAME "<<name<<" is "<<sum;
  return 0;
}
```
