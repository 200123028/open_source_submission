<h2>Cpp Code for Sum of ASCII values of Name</h2>

``` 

#include<bits/stdc++.h>

using namespace std;

int main(){
	int sum = 0;
	string name;
	cin>>name;
	for(int i = 0; name[i] != '\0'; i++){
		if(name[i] == ' ')
			continue;
		else
			sum += name[i];
	}
	
	cout<<"Sum of values of ASCII of name "<<sum<<endl;
	return 0;
}
```
