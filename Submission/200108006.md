### *__C++ code to calculate the ASCII sum of my name:__*
```c++
#include<bits/stdc++.h>

using namespace std;

int main(void){
    string myname = "Akshay Khandelwal";

    long long ascii_sum = 0;
    for (auto ch: myname){
        if (ch != ' ') ascii_sum += ch;
    }

    cout << "Final answer = " << ascii_sum << "\n";

    return 0;
}
```
